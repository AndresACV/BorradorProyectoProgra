create database ProyectoI;
use ProyectoI;
create table Empleado(
    cedula varchar(10) not null,
    nombre varchar(30) not null, 
    telefono varchar(10) not null, 
    salarioBase double not null, 
    salarioTotal double not null,
    sucursal varchar(3) not null,
    primary key(cedula)
);
create table Sucursal(
    codigo varchar(3) not null, 
    referencia varchar(30) not null,
    direccion varchar(30) not null, 
    porcentajeZonaje double not null,
    x int not null,
    y int not null, 
    primary key(codigo)
);
ALTER TABLE empleado add foreign key(sucursal) references Sucursal(codigo);
